springbok_db for MongoDB

Example:

class User extends Model {
  static final $ = new Model$<User>(User);

  Id id;
  String firstName;
  String lastName;
  int age;
}

main() {
  User.$.init().then((_) {
    User.$.findOne().then((User user) {
      print(user);
    });
  });
}
